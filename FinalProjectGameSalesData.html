df = pd.DataFrame({"Int": [78, 79, 82, 85, 91, 97, 108, 122, 138, 152]})
df.index = ["2010", "2011", "2012", "2013", "2014", "2015", "2016", "2017", "2018", "2019"]

df = df.sort_values(by="Int", ascending=False)
df["cumpercentage"] = df["Int"].cumsum()/df["Int"].sum()*100

fix, ax = plt.subplots(figsize=(20,10))
ax.bar(df.index, df["Int"], color = "C1")
ax2 = ax.twinx()
ax2.plot(df.index, df["cumpercentage"], color = "C2", marker="D", ms=7)
ax2.yaxis.set_major_formatter(PercentFormatter())

ax.tick_params(axis="y", colors="C2")
ax2.tick_params(axis="y", colors="C7")
plt.title("Gaming Industry Worldwide Revenue (2010 - 2019)")
plt.xlabel("Total Revenue in Billions")
plt.ylabel("Cumulative Percentage")
plt.show()
